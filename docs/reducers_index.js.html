import { combineReducers } from 'redux';
import { reducer as formReducer } from 'redux-form';

/**
 * Reducer for receiving search results
 * 
 * @param {Array} state
 * @param {Object} action
 * @returns {Array} nextState
 */
const searchResults = (state = [], action) => {
  switch (action.type) {
    case 'RECIEVE_FILMS':
      return action.data;
    default:
      return state;
  }
};

/**
 * Reducer for receiving film details
 * 
 * @param {Object} state
 * @param {Object} action
 * @returns {Object} nextState
 */
const detailItem = (state = {}, action) => {
  switch (action.type) {
    case 'RECIEVE_FILM_DETAIL':
      return action.data;
    default:
      return state;
  }
};

/**
 * Reducer for updating sortBy
 * 
 * @param {string} state
 * @param {Object} action
 * @returns {string} nextState
 */
const sortBy = (state = '', action) => {
  switch (action.type) {
    case 'UPDATE_SORT_BY':
      return action.sortBy;
    default:
      return state;
  }
};

/**
 * Reducer for receiving searchBy
 * 
 * @param {String} state
 * @param {Object} action
 * @returns {String} nextState
 */
const searchBy = (state = 'movie', action) => {
  switch (action.type) {
    case 'UPDATE_SEARCH_BY':
      return action.searchBy;
    default:
      return state;
  }
};

/**
 * Combines all reducers and redux-form reducer
 */
export default combineReducers({
  searchResults,
  sortBy,
  searchBy,
  detailItem,
  form: formReducer,
});
